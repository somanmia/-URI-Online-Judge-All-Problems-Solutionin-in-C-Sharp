using System;
namespace _2140
{
    class URI
    {
        static void Main(string[] args)
        {
            String userinput;
            int i, ck,between,M,N;
            int  []arr = {2, 5, 10, 20, 50 , 100 };
            while (true)
            {
                userinput = Console.ReadLine();
                String  []data = userinput.Split(' ');
                M = Convert.ToInt32(data[0]);
                N = Convert.ToInt32(data[1]);
                if (M == 0 && N == 0) break;
                between = N-M;
                if (between >= 100) between -= 100;
                else if (between >= 50) between -= 50;
                else if (between >= 20) between -= 20;
                else if (between >= 10) between -= 10;
                else if (between >= 5) between -= 5;
                else if (between >= 2) between -= 2;
                ck = 0;
                for (i = 0; i <6; i++)
                {
                    if (arr[i] == between)
                    {
                        ck++;
                        break;
                    }
                }
                if (ck > 0)
                {
                    Console.WriteLine("possible");
                }
                else
                {
                    Console.WriteLine("impossible");
                }
            }
            Console.ReadKey();
        }
    }
}
